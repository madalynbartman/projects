procedure fizzbuzz
    For i := 1 to 100 dp
        set print_number to true;
        If i is divisible by 3 then
            print "Fizz";
            set print_number to false;
        If i is divisible by 5 then 
            print "Buzz";
            set print_number to false;
        If print_number, print i;
        print a newline;
    end
